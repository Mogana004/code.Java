```java
import java.util.Scanner;
class printNumber {
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        
        // For loop initialized from 1 to N 
        for (int  i = 1; i<=N ; i++){
            System.out.println(i);
        }
        
        
    }
}
```
